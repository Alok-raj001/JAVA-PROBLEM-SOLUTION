public class Main
{
  
    static int count(int n)   {
            return helper(n,0);
            
        }
         private static int helper(int n,int c){  //We can make this method as public too.
      if(n==0){
          return c;
      }
       int rem=n%10;
        if(rem==0) {
            return helper(n/10,c+1);
        }
        return helper(n/10,c);
    }
	public static void main(String[] args) {
	  int sum= count(120300040);
		System.out.println(sum);
	}
}
